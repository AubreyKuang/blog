	public int search(int[] nums, int target) {
        int right = nums.length - 1, left = 0;
        while (left <= right){
            int mid = left + (right - left) / 2;
            int num = nums[mid];

            if (num == target) {
                return mid;
            }
            else if (target > num){
                left = mid + 1;
            }else{
                right = mid - 1;
            }
        }
        return -1;
    }

// 猜数字大小
// -1：选出的数字比猜的数字小 pick < num
// 1：选出的数字比猜的数字大 pick > num
// 0：选出的数字和猜的数字一样。pick == num

public class Solution extends GuessGame {
    public int guessNumber(int n) {
        int left = 1, right = n;
        while (left < right){
            int mid = left + (right - left) / 2;
            
            switch (guess(mid)){
                case -1:
                right = mid - 1;
                break;

                case 1:
                left = mid + 1;
                break;

                case 0 :
                right = mid;
                break;
            }
        }
    return right;
    }
}
