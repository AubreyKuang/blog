public int pivotIndex(int[] nums) {
       int Leftsum = 0, Rightsum = Arrays.stream(nums).sum();
       for (int i = 0; i < nums.length; i++){

        Rightsum -= nums[i];
        if (Leftsum == Rightsum)
            return i;
        
        Leftsum += nums[i];

        }
        return -1;
    }
